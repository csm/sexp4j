001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004import org.apache.commons.codec.binary.Base64;
005
006import java.io.EOFException;
007import java.io.IOException;
008import java.io.InputStream;
009import java.nio.ByteBuffer;
010import java.util.BitSet;
011import java.util.LinkedList;
012
013/**
014 * A streaming parser for the "advanced" encoding.
015 */
016public class AdvancedStreamingParser extends StreamingParser {
017    private static enum State {
018        ConsumeWhitespace,
019        StartingSymbolOrVerbatim,
020        ConsumeSymbol,
021        ConsumeQuotedString,
022        ConsumeHex,
023        ConsumeBase64
024    }
025
026    private int listDepth = 0;
027    private State state = State.ConsumeWhitespace;
028    private final LinkedList<Integer> tokens;
029    private Optional<Integer> quotedStringLength = Optional.absent();
030    private boolean consumingDisplayHint = false;
031    private byte[] displayHint = null;
032
033    public AdvancedStreamingParser(InputStream input) {
034        super(input);
035        tokens = new LinkedList<>();
036    }
037
038    @Override
039    public void parse() throws IOException {
040        boolean keepGoing = false;
041        do {
042            switch (state) {
043                case ConsumeWhitespace:
044                    keepGoing = consumeWhitespace();
045                    break;
046
047                case StartingSymbolOrVerbatim:
048                    keepGoing = consumeStartingSymbolOrVerbatim();
049                    break;
050
051                case ConsumeBase64:
052                    keepGoing = consumeBase64();
053                    break;
054
055                case ConsumeHex:
056                    keepGoing = consumeHex();
057                    break;
058
059                case ConsumeQuotedString:
060                    keepGoing = consumeQuotedString();
061                    break;
062
063                case ConsumeSymbol:
064                    keepGoing = consumeSymbol();
065                    break;
066            }
067        } while (keepGoing);
068    }
069
070    private boolean consumeWhitespace() throws IOException {
071        int b = input.read();
072        if (b == '(') {
073            onListBegin();
074            listDepth++;
075        }
076        else if (b == ')') {
077            if (listDepth == 0)
078                throw new ParseException("extra ')' in input");
079            onListEnd();
080            listDepth--;
081        }
082        else if (Character.isWhitespace(b))
083            return true;
084        else if (b == '[') {
085            if (consumingDisplayHint)
086                throw new ParseException("nested display-hints");
087            consumingDisplayHint = true;
088            return true;
089        }
090        else if (b == ']') {
091            if (!consumingDisplayHint)
092                throw new ParseException("unexpected token ']'");
093            displayHint = new byte[0];
094            consumingDisplayHint = false;
095        }
096        else if (b == '#')
097            state = State.ConsumeHex;
098        else if (b == '|')
099            state = State.ConsumeBase64;
100        else if (b == '"')
101            state = State.ConsumeQuotedString;
102        else if (Character.isDigit(b)) {
103            tokens.addLast(b);
104            state = State.StartingSymbolOrVerbatim;
105        }
106        else if (Character.isLetter(b)) {
107            tokens.addLast(b);
108            state = State.ConsumeSymbol;
109        }
110        else if (b < 0) {
111            if (listDepth > 0)
112                throw new ParseException("unbalanced parentheses");
113            return false;
114        }
115        else
116            throw new ParseException("unexpected token %c", (char) b);
117        return true;
118    }
119
120    private boolean consumeStartingSymbolOrVerbatim() throws IOException {
121        while (true) {
122            int b = input.read();
123            if (Character.isDigit(b)) {
124                tokens.push(b);
125            }
126            else if (Character.isLetter(b)) {
127                state = State.ConsumeSymbol;
128                break;
129            }
130            else if (b == ':') {
131                int length = makeLength();
132                byte[] buffer = new byte[length];
133                int read = input.read(buffer);
134                if (read < length)
135                    throw new EOFException();
136                if (consumingDisplayHint) {
137                    displayHint = buffer;
138                    consumeWhitespaceUntil(']');
139                    consumingDisplayHint = false;
140                } else {
141                    onAtom(buffer, Optional.fromNullable(displayHint));
142                    displayHint = null;
143                }
144                state = State.ConsumeWhitespace;
145                break;
146            }
147            else if (b == '"') {
148                quotedStringLength = Optional.of(makeLength());
149                state = State.ConsumeQuotedString;
150                break;
151            }
152            else if (Character.isWhitespace(b)) {
153                makeSymbol();
154                state = State.ConsumeWhitespace;
155                break;
156            }
157            else if (b < 0) {
158                throw new ParseException("end-of-file reading symbol");
159            }
160            else
161                throw new ParseException("unexpected token: %c", (char) b);
162        }
163        return true;
164    }
165
166    private void consumeWhitespaceUntil(int stopchar) throws IOException {
167        while (true) {
168            int ch = input.read();
169            if (ch == stopchar)
170                break;
171            else if (Character.isWhitespace(ch))
172                continue;
173            else if (ch == -1)
174                throw new EOFException();
175            else
176                throw new ParseException("unexpected token %02x", ch);
177        }
178    }
179
180    private int makeLength() throws ParseException {
181        int length = 0;
182        while (!tokens.isEmpty()) {
183            int i = tokens.removeFirst();
184            int r = length * 10;
185            int next = (i - '0');
186            if (r < length || r + next < length)
187                throw new ParseException("integer overflow: atom length is greater than 2^31-1");
188            length = r + next;
189        }
190        return length;
191    }
192
193    private void makeSymbol() throws IOException {
194        byte[] buffer = new byte[tokens.size()];
195        int i = 0;
196        while (!tokens.isEmpty()) {
197            buffer[i++] = tokens.removeFirst().byteValue();
198        }
199        if (consumingDisplayHint) {
200            displayHint = buffer;
201            // note, don't need to consume whitespace until ']', because
202            // we already got it.
203            consumingDisplayHint = false;
204        } else {
205            onAtom(buffer, Optional.fromNullable(displayHint));
206            displayHint = null;
207        }
208    }
209
210    private boolean consumeSymbol() throws IOException {
211        while (true) {
212            int b = input.read();
213            if (Character.isLetterOrDigit(b)) {
214                tokens.addLast(b);
215            }
216            else if (Character.isWhitespace(b)) {
217                makeSymbol();
218                state = State.ConsumeWhitespace;
219                break;
220            }
221            else if (b == '(') {
222                makeSymbol();
223                onListBegin();
224                listDepth++;
225                state = State.ConsumeWhitespace;
226                break;
227            }
228            else if (b == ')') {
229                if (listDepth == 0)
230                    throw new ParseException("unbalanced parentheses");
231                makeSymbol();
232                onListEnd();
233                listDepth--;
234                state = State.ConsumeWhitespace;
235                break;
236            }
237            else if (b == ']' && consumingDisplayHint) {
238                makeSymbol();
239                state = State.ConsumeWhitespace;
240                break;
241            }
242            else if (b < 0) {
243                if (listDepth > 0)
244                    throw new EOFException();
245                makeSymbol();
246                state = State.ConsumeWhitespace;
247                return false;
248            }
249            else {
250                throw new ParseException("unexpected token: %c", (char) b);
251            }
252        }
253        return true;
254    }
255
256    private boolean consumeBase64() throws IOException {
257        ByteBuffer buffer = ByteBuffer.allocate(16);
258        while (true) {
259            int b = input.read();
260            if (Character.isLetterOrDigit(b) || b == '+' || b == '/' || b == '=') {
261                if (buffer.remaining() == 0)
262                    buffer = grow(buffer);
263                buffer.put((byte) b);
264            }
265            else if (Character.isWhitespace(b)) {
266                continue;
267            }
268            else if (b == '|') {
269                buffer.flip();
270                byte[] bytes = new byte[buffer.remaining()];
271                buffer.get(bytes);
272                bytes = Base64.decodeBase64(bytes);
273                if (consumingDisplayHint) {
274                    displayHint = bytes;
275                    consumeWhitespaceUntil(']');
276                    consumingDisplayHint = false;
277                } else {
278                    onAtom(bytes, Optional.fromNullable(displayHint));
279                    displayHint = null;
280                }
281                state = State.ConsumeWhitespace;
282                break;
283            }
284            else if (b < 0) {
285                throw new EOFException();
286            }
287            else {
288                throw new ParseException("unexpected token: 0x%x", b);
289            }
290        }
291        return true;
292    }
293
294    private int hexchar(int c) {
295        if (c >= '0' && c <= '9')
296            return c - '0';
297        else if (c >= 'a' && c <= 'f')
298            return (c - 'a') + 10;
299        else if (c >= 'A' && c <= 'F')
300            return (c - 'A') + 10;
301        else
302            throw new IllegalArgumentException("unexpected char: " + (char) c);
303    }
304
305    private boolean consumeHex() throws IOException {
306        StringBuilder buffer = new StringBuilder();
307        while (true) {
308            int b = input.read();
309            if (Character.isDigit(b) || (b >= 'a' && b <= 'f') || (b >= 'A' && b <= 'F')) {
310                buffer.append((char) b);
311            }
312            else if (Character.isWhitespace(b)) {
313                continue;
314            }
315            else if (b == '#') {
316                if (buffer.length() % 2 == 1)
317                    buffer.insert(0, '0');
318                byte[] result = new byte[buffer.length() / 2];
319                for (int i = 0, j = 0; i + 1 < buffer.length() && j < result.length; i += 2, j++) {
320                    result[j] = (byte) ((hexchar(buffer.charAt(i)) << 4) | hexchar(buffer.charAt(i + 1)));
321                }
322                if (consumingDisplayHint) {
323                    displayHint = result;
324                    consumeWhitespaceUntil(']');
325                    consumingDisplayHint = false;
326                } else {
327                    onAtom(result, Optional.fromNullable(displayHint));
328                    displayHint = null;
329                }
330                state = State.ConsumeWhitespace;
331                break;
332            }
333            else if (b < 0) {
334                throw new EOFException();
335            }
336            else {
337                throw new ParseException("unexpected token: 0x%x", b);
338            }
339        }
340        return true;
341    }

343    private static final BitSet quotedStringChars = new BitSet('}');
344    static {
345        quotedStringChars.set('0', '9' + 1);
346        quotedStringChars.set('a', 'z' + 1);
347        quotedStringChars.set('A', 'Z' + 1);
348        quotedStringChars.set(' ');
349        quotedStringChars.set('\t');
350        quotedStringChars.set('\n');
351        quotedStringChars.set('\r');
352        quotedStringChars.set('-');
353        quotedStringChars.set('.');
354        quotedStringChars.set('/');
355        quotedStringChars.set('_');
356        quotedStringChars.set(':');
357        quotedStringChars.set('*');
358        quotedStringChars.set('+');
359        quotedStringChars.set('=');
360        quotedStringChars.set('(');
361        quotedStringChars.set(')');
362        quotedStringChars.set('[');
363        quotedStringChars.set(']');
364        quotedStringChars.set('{');
365        quotedStringChars.set('}');
366        quotedStringChars.set('|');
367        quotedStringChars.set('#');
368        quotedStringChars.set('&');
369    }
370
371    private boolean consumeQuotedString() throws IOException {
372        Optional<Integer> len = quotedStringLength;
373        quotedStringLength = Optional.absent();
374        ByteBuffer buffer = ByteBuffer.allocate(len.or(16));
375        LinkedList<Integer> peek = new LinkedList<>();
376        while (true) {
377            int b;
378            if (peek.isEmpty())
379                b = input.read();
380            else
381                b = peek.removeFirst();
382            if (quotedStringChars.get(b)) {
383                buffer = doPut(buffer, (byte) b);
384            }
385            else if (b == '\\') {
386                int e1 = input.read();
387                switch (e1) {
388                    case 'b':
389                        buffer = doPut(buffer, (byte) '\b');
390                        break;
391                    case 't':
392                        buffer = doPut(buffer, (byte) '\t');
393                        break;
394                    case 'v':
395                        buffer = doPut(buffer, (byte) 0xb);
396                        break;
397                    case 'n':
398                        buffer = doPut(buffer, (byte) '\n');
399                        break;
400                    case 'r':
401                        buffer = doPut(buffer, (byte) '\r');
402                        break;
403                    case '"':
404                        buffer = doPut(buffer, (byte) '"');
405                        break;
406                    case '\'':
407                        buffer = doPut(buffer, (byte) '\'');
408                        break;
409                    case '\\':
410                        buffer = doPut(buffer, (byte) '\\');
411                        break;
412                    case '0':
413                    case '1':
414                    case '2':
415                    case '3':
416                    case '4':
417                    case '5':
418                    case '6':
419                    case '7': {
420                        int e2 = input.read();
421                        int e3 = input.read();
422                        if (e2 < '0' || e2 > '7' || e3 < '0' || e3 > '7')
423                            throw new ParseException("invalid octal value in quoted string");
424                        int value = ((e1 - '0') << 6) | ((e2 - '0') << 3) | (e3 - '0');
425                        buffer = doPut(buffer, (byte) value);
426                        break;
427                    }
428                    case 'x': {
429                        int e2 = input.read();
430                        int e3 = input.read();
431                        int value = (hexchar(e2) << 4) | (hexchar(e3));
432                        buffer = doPut(buffer, (byte) value);
433                        break;
434                    }
435                    case '\r': {
436                        int e2 = input.read();
437                        if (e2 != '\n')
438                            peek.addLast(e2);
439                        break; // ignore these characters
440                    }
441                    case '\n': {
442                        int e2 = input.read();
443                        if (e2 != '\r')
444                            peek.addLast(e2);
445                        break; // ignore these characters
446                    }
447                    default:
448                        throw new ParseException("invalid escape character: %c (0x%x)", (char) e1, e1);
449                }
450            }
451            else if (b == '"') {
452                buffer.flip();
453                if (len.isPresent() && len.get() != buffer.remaining())
454                    throw new ParseException("quoted string length did not match explicit length");
455                byte[] bytes = new byte[buffer.remaining()];
456                buffer.get(bytes);
457                if (consumingDisplayHint) {
458                    displayHint = bytes;
459                    consumeWhitespaceUntil(']');
460                    consumingDisplayHint = false;
461                } else {
462                    onAtom(bytes, Optional.fromNullable(displayHint));
463                    displayHint = null;
464                }
465                state = State.ConsumeWhitespace;
466                break;
467            }
468            else if (b < 0) {
469                throw new EOFException();
470            }
471            else {
472                throw new ParseException("invalid character in quoted string: %c (0x%x)", (char) b, b);
473            }
474        }
475        return true;
476    }
477
478    private ByteBuffer doPut(ByteBuffer buffer, byte value) {
479        if (!buffer.hasRemaining())
480            buffer = grow(buffer);
481        buffer.put(value);
482        return buffer;
483    }
484
485    private ByteBuffer grow(ByteBuffer buffer) {
486        int newLength;
487        buffer.flip();
488        if (buffer.remaining() > 4096)
489            newLength = buffer.remaining() + 1024;
490        else
491            newLength = buffer.remaining() * 2;
492        ByteBuffer newBuffer = ByteBuffer.allocate(newLength);
493        newBuffer.put(buffer);
494        return newBuffer;
495    }
