001package org.metastatic.sexp4j;
002
003import java.io.IOException;
004
005/**
006 * An s-expression writer.
007 */
008public interface Writer {
009    /**
010     * Write an atom.
011     *
012     * <p>It is invalid to write an atom if a single atom has already
013     * been written, or if a list in its entirety has been written
014     * previously.</p>
015     *
016     * @param atom The atom to write.
017     * @return The number of bytes written.
018     * @throws NullPointerException if the argument is null.
019     * @throws WriteException If an illegal write state occurs.
020     * @throws IOException If an IO exception occurs.
021     */
022    int writeAtom(Atom atom) throws IOException;
023
024    /**
025     * Write a list.
026     *
027     * <p>It is invalid to write a list if a single atom was written
028     * as the only value, or if a complete list was written previously.
029     * </p>
030     *
031     * @param list The list to write.
032     * @return The number of bytes written.
033     * @throws NullPointerException if the argument is null.
034     * @throws WriteException If an illegal write state occurs.
035     * @throws IOException If an IO exception occurs.
036     */
037    int writeList(ExpressionList list) throws IOException;
038
039    /**
040     * Begin a list. You don't need to call this along with {@link #writeList(ExpressionList)},
041     * this method exists to support writing expressions without building them
042     * as {@link org.metastatic.sexp4j.ExpressionList} instances first.
043     *
044     * @throws WriteException If it is illegal to begin a list in the current state.
045     * @throws IOException If an IO exception occurs.
046     */
047    void beginList() throws IOException;
048
049    /**
050     * End a list. You should call this if you use the {@link #beginList()} method
051     * to start a list.
052     *
053     * @throws WriteException if it is illegal to end a list in the current state.
054     * @throws IOException If an IO exception occurs.
055     */
056    void endList() throws IOException;
057
058    /**
059     * Write an expression.
060     *
061     * @param expression The expression to write.
062     * @return The number of bytes written.
063     * @throws NullPointerException if the argument is null.
064     * @throws IllegalArgumentException if the argument is not a {@link Atom}
065     *         or a {@link ExpressionList}.
066     * @throws WriteException If an illegal write state occurs.
067     * @throws IOException If an IO exception occurs.
068     * @see #writeAtom(Atom)
069     * @see #writeList(ExpressionList)
070     */
071    int writeExpression(Expression expression) throws IOException;
072}
