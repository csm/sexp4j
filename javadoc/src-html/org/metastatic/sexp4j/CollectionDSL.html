001package org.metastatic.sexp4j;
002
003import com.google.common.base.Preconditions;
004
005import java.util.*;
006
007/**
008 * Created by cmarshall on 12/8/14.
009 */
010public class CollectionDSL {
011    public static Map map(Object... keyValues) {
012        Preconditions.checkArgument(keyValues.length % 2 == 0);
013        Map ret = new LinkedHashMap(keyValues.length / 2);
014        for (int i = 0; i < keyValues.length; i += 2) {
015            ret.put(keyValues[i], keyValues[i + 1]);
016        }
017        return ret;
018    }
019
020    public static Set set(Object... values) {
021        Map ret = new LinkedHashMap(values.length);
022        for (Object value : values)
023            ret.put(value, null);
024        return ret.keySet();
025    }
026
027    public static List list(Object... values) {
028        return Arrays.asList(values);
029    }
030}
