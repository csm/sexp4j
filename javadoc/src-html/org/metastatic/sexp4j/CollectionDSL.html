001package org.metastatic.sexp4j;
002
003import com.google.common.base.Preconditions;
004
005import java.util.*;
006
007/**
008 * DSL for creating collections.
009 */
010public class CollectionDSL {
011    /**
012     * Create a linked hash map with the given key/values.
013     *
014     * @param keyValues A sequence of pairs of keys and values.
015     * @return The new map.
016     */
017    public static Map map(Object... keyValues) {
018        Preconditions.checkArgument(keyValues.length % 2 == 0);
019        Map ret = new LinkedHashMap(keyValues.length / 2);
020        for (int i = 0; i < keyValues.length; i += 2) {
021            ret.put(keyValues[i], keyValues[i + 1]);
022        }
023        return ret;
024    }
025
026    /**
027     * Return a set (hashed, but preserving order) of the given values.
028     *
029     * @param values The values.
030     * @return The set.
031     */
032    public static Set set(Object... values) {
033        Map ret = new LinkedHashMap(values.length);
034        for (Object value : values)
035            ret.put(value, null);
036        return ret.keySet();
037    }
038
039    /**
040     * Return a list of objects.
041     *
042     * @param values The values.
043     * @return The list.
044     */
045    public static List list(Object... values) {
046        return Arrays.asList(values);
047    }
048}
