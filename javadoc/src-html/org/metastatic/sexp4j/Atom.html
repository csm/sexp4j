016/**
017 * An atom; that is, a sequence of bytes, with an optional display hint.
018 */
019public class Atom implements Cloneable, Expression
020{
021    private final byte[] bytes;
022    private final Optional<DisplayHint> displayHint;
023
024    /**
025     * Create an atom with the given bytes.
026     *
027     * @param bytes The atom bytes; this value is copied.
028     * @throws java.lang.NullPointerException If the argument is null.
029     */
030    public Atom(byte[] bytes) {
031        Preconditions.checkNotNull(bytes);
032        this.bytes = bytes.clone();
033        displayHint = Optional.absent();
034    }
035
036    /**
037     * @param code
038     * @param bytes
039     */
040    @Deprecated
041    public Atom(byte code, byte[] bytes) {
042        Preconditions.checkNotNull(bytes);
043        this.bytes = new byte[bytes.length + 1];
044        this.bytes[0] = code;
045        System.arraycopy(bytes, 0, this.bytes, 1, bytes.length);
046        displayHint = Optional.absent();
047    }
048
049    /**
050     * Create an atom with a subsequence of a byte array.
051     *
052     * @param bytes The byte array.
053     * @param offset The offset of the bytes to read.
054     * @param length The number of bytes to read.
055     * @throws java.lang.IllegalArgumentException If the offset is negative, or if the offset plus the length is larger than the number of bytes in the array.
056     * @throws java.lang.NullPointerException If the byte array is null.
057     */
058    public Atom(byte[] bytes, int offset, int length) {
059        Preconditions.checkNotNull(bytes);
060        Preconditions.checkArgument(offset >= 0);
061        Preconditions.checkArgument(offset + length <= bytes.length);
062        this.bytes = new byte[length];
063        System.arraycopy(bytes, offset, this.bytes, 0, length);
064        displayHint = Optional.absent();
065    }
066
067    private Atom(byte[] bytes, DisplayHint hint) {
068        this.bytes = bytes;
069        this.displayHint = Optional.of(hint);
070    }

348    private static final BitSet symbolChars = new BitSet(256);
349    private static final BitSet quotedStringChars = new BitSet(256);
350
351    static {
352        symbolChars.set('a', 'z' + 1);
353        symbolChars.set('A', 'Z' + 1);
354        symbolChars.set('0', '9' + 1);
355
356        quotedStringChars.set(0x20);
357        quotedStringChars.set('!');
358        quotedStringChars.set('#', '~' + 1);
359    }
360
361    /**
362     * Tell if this atom can be encoded as a symbol; that is, only
363     * contains alphanumeric characters.
364     *
365     * @return True if this may be encoded as a symbol.
366     */
367    public boolean canBeSymbol() {
368        if (bytes.length == 0)
369            return false;
370        for (byte b : bytes) {
371            if (!symbolChars.get(b & 0xFF))
372                return false;
373        }
374        return true;
375    }
376
377    /**
378     * Tell if this atom can be encoded as a quoted string.
379     *
380     * @return True if this may be encoded as a quoted string.
381     */
382    public boolean canBeQuotedString() {
383        for (byte b : bytes) {
384            if (!quotedStringChars.get(b & 0xff))
385                return false;
386        }
387        return true;
388    }

390    private byte hexByte(int i) {
391        if (i >= 0 && i < 10) {
392            return (byte) ('0' + i);
393        }
394        return (byte) ('a' + (i - 10));
395    }
396
397    /**
398     * Return the value of this atom, encoded as a byte array of
399     * hexadecimal ASCII characters.
400     *
401     * @return The hex representation.
402     */
403    public byte[] asHexBytes() {
404        byte[] result = new byte[bytes.length * 2];
405        int i = 0;
406        for (byte b : bytes) {
407            int x = b & 0xFF;
408            result[i++] = hexByte((x >> 4) & 0xf);
409            result[i++] = hexByte(x & 0xf);
410        }
411        return result;
412    }
