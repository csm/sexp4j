001package org.metastatic.sexp4j.mapper;
002
003import java.math.BigDecimal;
004import java.math.BigInteger;
005import java.util.*;
006
007import com.google.common.annotations.Beta;
008import org.metastatic.sexp4j.Atom;
009import org.metastatic.sexp4j.Expression;
010import org.metastatic.sexp4j.ExpressionList;
011
012/**
013 * A "simple" mapper between Java objects and expressions.
014 *
015 * <p>The definition of a "simple" object is:</p>
016 *
017 * <ul>
018 *     <li>Any "primitive" type. These will be encoded as atoms, with one-character display hints, as follows:
019 *     <ul>
020 *         <li>boolean, a byte 0 or 1, with display hint "z".</li>
021 *         <li>byte, one byte, with display hint "b".</li>
022 *         <li>short, two bytes (multi-byte primitives are in big endian order), display hint "s".</li>
023 *         <li>char, two bytes, display hint "c".</li>
024 *         <li>int, four bytes, display hint "i".</li>
025 *         <li>long, eight bytes, display hint "l".</li>
026 *         <li>float, four bytes, display hint "f".</li>
027 *         <li>double, eight bytes, display hint "d".</li>
028 *         <li>nulls, zero bytes, display hint "n".</li>
029 *         <li>Strings, utf-8 bytes, display hint "S".</li>
030 *         <li>byte arrays, as-is, display hint "B".</li>
031 *         <li>BigInteger, result of toByteArray, display hint "I".</li>
032 *         <li>BigDecimal, result of toString then encoded as UTF-8, display hint "D".</li>
033 *     </ul></li>
034 *     <li>A list of simpler objects. This is encoded as a list: a type code atom "l", then the sequence of simpler expressions.</li>
035 *     <li>A set of simpler objects. This is encoded as a list: a type code atom "s", then the sequence of simpler expressions.</li>
036 *     <li>A map of strings to any simpler type. This is encoded as a list: a type code atom "m", then a sequence of key/value expressions.
037 *     The key expression is always a "string", that is, an atom, without any display-hint; the value can be any expression.</li>
038 * </ul>
039 *
040 * <p>An example encoding:</p>
041 *
042 * <pre>
043(m aBool [z] #01#
044   aNull [n] 0:
045   aByte [b] #1f#
046   aShort [s] #007b#
047   anInt [i] #0001e240#
048   aLong [l] #00000002dfdc1c34#
049   aFloat [f] #4048f5c3#
050   aDouble [d] #400920c49ba5e354#
051   bytes [B] "just some bytes"
052   string [S] "just a string"
053   bigint [I] |EkmtJZTDfOsLJ4TEzgvzis5AjiEafKqyQwioLo8QAAAAAAAAAAAAAAAA|
054   bigdec [D] "3.1415929203"
055   list (l [S] just [S] some [S] items [S] in [S] a [S] list)
056   set (s [S] more [S] items [S] in [S] a [S] set [S] but [S] no [S] repeated)
057   map (m submaps [z] #01#))
058 * </pre>
059 */
060@Beta
061public class SimpleMapper {
062    public static enum Type {
063        Null((byte) 'n'),
064        Bool((byte) 'z'),
065        Byte((byte) 'b'),
066        Short((byte) 's'),
067        Char((byte) 'c'),
068        Int((byte) 'i'),
069        Long((byte) 'l'),
070        Float((byte) 'f'),
071        Double((byte) 'd'),
072        String((byte) 'S'),
073        BigInt((byte) 'I'),
074        BigDecimal((byte) 'D'),
075        Bytes((byte) 'B'),
076        List((byte) 'l'),
077        Set((byte) 's'),
078        Map((byte) 'm');
079
080        private Type(byte code) {
081            this.code = code;
082        }
083
084        private final byte code;
085    }
086
087    private Atom atom(byte code, byte[] bytes) {
088        return new Atom(code, bytes);
089    }
090
091    public Expression encode(Object o) throws MapperException {
092        if (o == null)
093            return new Atom(new byte[0]).withHint(Type.Null.code);
094        Class clazz = o.getClass();
095        if (clazz.equals(Boolean.class))
096            return new Atom(new byte[] { (boolean) o ? (byte) 1 : (byte) 0 }).withHint(Type.Bool.code);
097        if (clazz.equals(Byte.class))
098            return new Atom(new byte[] { (byte) o }).withHint(Type.Byte.code);
099        if (clazz.equals(Short.class))
100            return Atom.atom((short) o).withHint(Type.Short.code);
101        if (clazz.equals(Character.class))
102            return Atom.atom((char) o).withHint(Type.Char.code);
103        if (clazz.equals(Integer.class))
104            return Atom.atom((int) o).withHint(Type.Int.code);
105        if (clazz.equals(Long.class))
106            return Atom.atom((long) o).withHint(Type.Long.code);
107        if (clazz.equals(Float.class))
108            return Atom.atom((float) o).withHint(Type.Float.code);
109        if (clazz.equals(Double.class))
110            return Atom.atom((double) o).withHint(Type.Double.code);
111        if (clazz.equals(String.class))
112            return Atom.atom((String) o).withHint(Type.String.code);
113        if (clazz.equals(BigInteger.class))
114            return new Atom(((BigInteger) o).toByteArray()).withHint(Type.BigInt.code);
115        if (clazz.equals(BigDecimal.class))
116            return Atom.atom(o.toString()).withHint(Type.BigDecimal.code);
117        if (clazz.isArray() && clazz.getComponentType().equals(Byte.TYPE))
118            return new Atom((byte[]) o).withHint(Type.Bytes.code);
119        if (List.class.isAssignableFrom(clazz)) {
120            ExpressionList list = new ExpressionList(((List) o).size() + 1);
121            list.add(Atom.atom(Type.List.code));
122            for (Object e : ((List) o))
123                list.add(encode(e));
124            return list;
125        }
126        if (Set.class.isAssignableFrom(clazz)) {
127            ExpressionList list = new ExpressionList(((Set) o).size() + 1);
128            list.add(Atom.atom(Type.Set.code));
129            for (Object e : (Set) o)
130                list.add(encode(e));
131            return list;
132        }
133        if (Map.class.isAssignableFrom(clazz)) {
134            ExpressionList list = new ExpressionList(((Map) o).size() * 2 + 1);
135            list.add(Atom.atom(Type.Map.code));
136            for (Object e : ((Map) o).entrySet()) {
137                Map.Entry entry = (Map.Entry) e;
138                if (!(entry.getKey() instanceof String) || ((String) entry.getKey()).isEmpty())
139                    throw new MapperException("map keys must be nonempty strings");
140                list.add(Atom.atom((String) entry.getKey()));
141                list.add(encode(((Map.Entry) e).getValue()));
142            }
143            return list;
144        }
145        throw new MapperException("don't know how to encode a " + clazz);
146    }
147
148    public Object decode(Expression e) throws MapperException {
149        if (e instanceof Atom) {
150            byte code = '['; // if no explicit display hint, default to plain byte arrays.
151            if (((Atom) e).displayHint().isPresent()) {
152                code = ((Atom) e).displayHint().get().atom().typeCode();
153            }
154            switch (code) {
155                case 'z': return ((Atom) e).byteValue() != 0;
156                case 'n': return null;
157                case 'b': return ((Atom) e).byteValue();
158                case 's': return ((Atom) e).shortValue();
159                case 'c': return ((Atom) e).charValue();
160                case 'i': return ((Atom) e).intValue();
161                case 'l': return ((Atom) e).longValue();
162                case 'f': return ((Atom) e).floatValue();
163                case 'd': return ((Atom) e).doubleValue();
164                case 'S': return ((Atom) e).stringValue();
165                case 'B': return ((Atom) e).bytes();
166                case 'I': return ((Atom) e).bigIntegerValue();
167                case 'D': return ((Atom) e).bigDecimalValue();
168                default: throw new MapperException("invalid atom type code: %02x", ((Atom) e).typeCode());
169            }
170        }
171        else {
172            ExpressionList list = (ExpressionList) e;
173            Expression code = list.get(0);
174            if (!(code instanceof Atom))
175                throw new MapperException("expecting type code byte prefixing list");
176            switch (((Atom) code).typeCode()) {
177                case 'l': {
178                    List l = new ArrayList<>(list.size() - 1);
179                    for (Expression ex : list.subList(1, list.size()))
180                        l.add(decode(ex));
181                    return l;
182                }
183                case 's': {
184                    LinkedHashMap m = new LinkedHashMap(list.size() - 1);
185                    for (Expression ex : list.subList(1, list.size())) {
186                        Object v = decode(ex);
187                        m.put(v, v);
188                    }
189                    return m.keySet();
190                }
191                case 'm': {
192                    if (list.size() % 2 != 1) {
193                        throw new MapperException("maps must have an even number of items");
194                    }
195                    LinkedHashMap m = new LinkedHashMap(list.size() / 2 + 1);
196                    for (int i = 1; i + 1 < list.size(); i += 2) {
197                        Object key = list.get(i);
198                        if (!(key instanceof Atom) || ((Atom) key).length() == 0)
199                            throw new MapperException("map keys must be nonempty atoms");
200                        Object value = decode(list.get(i + 1));
201                        m.put(((Atom) key).stringValue(), value);
202                    }
203                    return m;
204                }
205                default:
206                    throw new MapperException("invalid type code: 0x%02x", ((Atom) code).typeCode());
207            }
208        }
209    }
210}
