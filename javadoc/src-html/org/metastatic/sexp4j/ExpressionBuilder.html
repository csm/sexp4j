001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004
005import java.io.UnsupportedEncodingException;
006import java.nio.charset.Charset;
007import java.util.LinkedList;
008
009public class ExpressionBuilder {
010    private final LinkedList<ExpressionList> stack = new LinkedList<>();
011    private Optional<Expression> first = Optional.absent();
012
013    public static ExpressionBuilder create() {
014        return new ExpressionBuilder();
015    }
016
017    public Expression build() {
018        if (first.isPresent()) {
019            stack.clear();
020            return first.get();
021        }
022        throw new IllegalStateException("nothing to build");
023    }
024
025    public ExpressionBuilder atom(Atom atom) {
026        if (first.isPresent() && first.get() instanceof Atom)
027            throw new IllegalStateException("first expression added was an atom");
028        if (stack.isEmpty())
029        {
030            if (first.isPresent())
031                throw new IllegalStateException("attempt to add atom outside of list");
032            first = Optional.<Expression> of(atom);
033        }
034        else
035            stack.peek().add(atom);
036        return this;
037    }
038
039    public ExpressionBuilder atom(byte[] atomBytes) {
040        return atom(new Atom(atomBytes));
041    }
042
043    public ExpressionBuilder atom(byte[] atomBytes, int offset, int length) {
044        return atom(new Atom(atomBytes, offset, length));
045    }
046
047    public ExpressionBuilder atom(String atomString) {
048        return atom(atomString.getBytes());
049    }
050
051    public ExpressionBuilder atom(String atomString, Charset charset) {
052        return atom(atomString.getBytes(charset));
053    }
054
055    public ExpressionBuilder atom(String atomString, String atomCharset) throws UnsupportedEncodingException {
056        return atom(atomString.getBytes(atomCharset));
057    }
058
059    public ExpressionBuilder beginList() {
060        if (first.isPresent() && first.get() instanceof Atom)
061            throw new IllegalStateException("first expression added was an atom");
062        ExpressionList list = new ExpressionList();
063        if (stack.isEmpty())
064        {
065            if (first.isPresent())
066                throw new IllegalStateException("attempt to begin list outside of list");
067            first = Optional.<Expression> of(list);
068        }
069        else
070            stack.peek().add(list);
071        stack.push(list);
072        return this;
073    }
074
075    public ExpressionBuilder endList()
076    {
077        if (stack.isEmpty())
078            throw new IllegalStateException("no list to end");
079        stack.pop();
080        return this;
081    }
082}
