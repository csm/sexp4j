001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004import com.google.common.base.Preconditions;
005
006import java.io.UnsupportedEncodingException;
007import java.nio.charset.Charset;
008import java.util.LinkedList;
009
010/**
011 * A builder interface for creating expressions.
012 */
013public class ExpressionBuilder {
014    private final LinkedList<ExpressionList> stack = new LinkedList<>();
015    private Optional<Expression> first = Optional.absent();
016
017    /**
018     * Create a new expression builder.
019     *
020     * @return The expression builder.
021     */
022    public static ExpressionBuilder create() {
023        return new ExpressionBuilder();
024    }
025
026    /**
027     * Build the expression.
028     *
029     * @return The built expression.
030     * @throws java.lang.IllegalStateException If there is nothing to build.
031     */
032    public Expression build() {
033        if (first.isPresent()) {
034            stack.clear();
035            return first.get();
036        }
037        throw new IllegalStateException("nothing to build");
038    }
039
040    /**
041     * Append an atom to this builder.
042     *
043     * @param atom The atom to append.
044     * @return This instance.
045     * @throws java.lang.NullPointerException If the argument is null.
046     * @throws java.lang.IllegalStateException If a single atom has already been added to this builder, or if this atom is being added outside of a list.
047     */
048    public ExpressionBuilder atom(Atom atom) {
049        Preconditions.checkNotNull(atom);
050        if (first.isPresent() && first.get() instanceof Atom)
051            throw new IllegalStateException("first expression added was an atom");
052        if (stack.isEmpty())
053        {
054            if (first.isPresent())
055                throw new IllegalStateException("attempt to add atom outside of list");
056            first = Optional.<Expression> of(atom);
057        }
058        else
059            stack.peek().add(atom);
060        return this;
061    }
062
063    /**
064     * Append an atom with the given bytes to this builder.
065     *
066     * @param atomBytes The atom bytes.
067     * @return This instance.
068     * @throws java.lang.NullPointerException If the argument is null.
069     * @throws java.lang.IllegalStateException If a single atom has already been added to this builder, or if this atom is being added outside of a list.
070     */
071    public ExpressionBuilder atom(byte[] atomBytes) {
072        return atom(new Atom(atomBytes));
073    }
074
075    /**
076     * Append an atom with the given bytes to this builder.
077     *
078     * @param atomBytes The atom bytes.
079     * @param offset The offset.
080     * @param length The length.
081     * @return This instance.
082     * @throws java.lang.NullPointerException If the argument is null.
083     * @throws java.lang.IllegalStateException If a single atom has already been added to this builder, or if this atom is being added outside of a list.
084     */
085    public ExpressionBuilder atom(byte[] atomBytes, int offset, int length) {
086        return atom(new Atom(atomBytes, offset, length));
087    }
088
089    /**
090     * Append an atom with the given string (in UTF-8) to this builder.
091     *
092     * @param atomString The string.
093     * @return This instance.
094     */
095    public ExpressionBuilder atom(String atomString) {
096        return atom(atomString.getBytes());
097    }
098
099    public ExpressionBuilder atom(String atomString, Charset charset) {
100        return atom(atomString.getBytes(charset));
101    }
102
103    public ExpressionBuilder atom(String atomString, String atomCharset) throws UnsupportedEncodingException {
104        return atom(atomString.getBytes(atomCharset));
105    }
106
107    /**
108     * Append the beginning of a new list to this builder.
109     *
110     * @return This instance.
111     */
112    public ExpressionBuilder beginList() {
113        if (first.isPresent() && first.get() instanceof Atom)
114            throw new IllegalStateException("first expression added was an atom");
115        ExpressionList list = new ExpressionList();
116        if (stack.isEmpty())
117        {
118            if (first.isPresent())
119                throw new IllegalStateException("attempt to begin list outside of list");
120            first = Optional.<Expression> of(list);
121        }
122        else
123            stack.peek().add(list);
124        stack.push(list);
125        return this;
126    }
127
128    /**
129     * Append the end of a list to this builder.
130     *
131     * @return This instance.
132     */
133    public ExpressionBuilder endList()
134    {
135        if (stack.isEmpty())
136            throw new IllegalStateException("no list to end");
137        stack.pop();
138        return this;
139    }
140}
