001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004
005/**
006 * A callback for parsing s-expressions.
007 */
008public interface StreamingParserCallback
009{
010    /**
011     * Called on the beginning of a list.
012     *
013     * @throws org.metastatic.sexp4j.ParseException If a list is not appropriate given the parser's current state.
014     */
015    void beginList() throws ParseException;
016
017    /**
018     * Called on the end of a list.
019     *
020     * @throws org.metastatic.sexp4j.ParseException If ending a list is not appropriate given the current state.
021     */
022    void endList() throws ParseException;
023
024    /**
025     * Called when parsing an atom.
026     *
027     * @param atom The atom.
028     * @param displayHint The display hint, if present.
029     * @throws org.metastatic.sexp4j.ParseException If an atom is not appropriate given the current parser state.
030     */
031    void onAtom(byte[] atom, Optional<byte[]> displayHint) throws ParseException;
032}
