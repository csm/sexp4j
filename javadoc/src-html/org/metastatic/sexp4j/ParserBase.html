001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004import com.google.common.base.Preconditions;
005
006import java.io.IOException;
007import java.util.LinkedList;
008
009/**
010 * Base adapter class for turning a {@link org.metastatic.sexp4j.StreamingParser}
011 * into a {@link org.metastatic.sexp4j.Parser}.
012 */
013public abstract class ParserBase implements Parser {
014    private Optional<Expression> root = Optional.absent();
015    private final LinkedList<ExpressionList> stack = new LinkedList<>();
016    private final StreamingParser stream;
017
018    private class Callback implements StreamingParserCallback {
019        @Override
020        public void beginList() throws ParseException {
021            ExpressionList newList = new ExpressionList();
022            if (!stack.isEmpty())
023                stack.peek().add(newList);
024            else
025            {
026                if (root.isPresent())
027                    throw new ParseException("found multiple root values");
028                root = Optional.<Expression> of(newList);
029            }
030            stack.push(newList);
031        }
032
033        @Override
034        public void endList() throws ParseException {
035            if (stack.isEmpty())
036                throw new ParseException("extraneous end list");
037            stack.pop();
038        }
039
040        @Override
041        public void onAtom(byte[] atom, Optional<byte[]> displayHint) throws ParseException {
042            Atom a = new Atom(atom);
043            if (displayHint.isPresent())
044                a = a.withHint(new Atom(displayHint.get()));
045            if (stack.isEmpty()) {
046                if (root.isPresent())
047                    throw new ParseException("found multiple root values");
048                root = Optional.<Expression> of(a);
049            }
050            else
051                stack.peek().add(a);
052        }
053    }
054
055    /**
056     * Create a new parser adapter.
057     *
058     * @param parser The streaming parser.
059     * @throws java.lang.NullPointerException If the argument is null.
060     */
061    protected ParserBase(StreamingParser parser) {
062        Preconditions.checkNotNull(parser);
063        stream = parser;
064        stream.addCallback(new Callback());
065    }
066
067    @Override
068    public Expression parse() throws IOException {
069        stream.parse();
070        return root.orNull();
071    }
072}
