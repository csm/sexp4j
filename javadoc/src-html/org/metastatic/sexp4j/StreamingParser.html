001package org.metastatic.sexp4j;
002
003import com.google.common.base.Optional;
004import com.google.common.base.Preconditions;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * Base class for a streaming parser.
013 *
014 * <p>To use a streaming parser, you register one or more
015 * {@link org.metastatic.sexp4j.StreamingParserCallback} instances
016 * with this object, which will be called as expressions are read
017 * from the underlying input stream.</p>
018 */
019public abstract class StreamingParser
020{
021    private final List<StreamingParserCallback> callbacks;
022    protected final InputStream input;
023
024    /**
025     * Construct a new parser.
026     *
027     * @param input The input stream to parse.
028     * @throws java.lang.NullPointerException If the argument is null.
029     */
030    protected StreamingParser(InputStream input)
031    {
032        Preconditions.checkNotNull(input);
033        this.input = input;
034        this.callbacks = new ArrayList<>();
035    }
036
037    /**
038     * Parse the input, calling the callbacks as expressions are read.
039     *
040     * @throws org.metastatic.sexp4j.ParseException If the input is invalid.
041     * @throws IOException If an IO exception occurs.
042     */
043    public abstract void parse() throws IOException;
044
045    /**
046     * Add a callback to this parser.
047     *
048     * @param callback The callback to add.
049     * @throws java.lang.NullPointerException If the argument is null.
050     */
051    public final void addCallback(StreamingParserCallback callback)
052    {
053        Preconditions.checkNotNull(callback);
054        if (!callbacks.contains(callback))
055            callbacks.add(callback);
056    }
057
058    /**
059     * Remove a callback from this parser.
060     *
061     * @param callback The callback to remove.
062     */
063    public final void removeCallback(StreamingParserCallback callback) {
064        callbacks.remove(callback);
065    }
066
067    /**
068     * Called when a list begins.
069     *
070     * @throws ParseException If a list is not appropriate given the current parser state.
071     */
072    protected final void onListBegin() throws ParseException
073    {
074        for (StreamingParserCallback callback : callbacks)
075            callback.beginList();
076    }
077
078    /**
079     * Called when a list ends.
080     *
081     * @throws ParseException If a list end is not appropriate given the current parser state.
082     */
083    protected final void onListEnd() throws ParseException
084    {
085        for (StreamingParserCallback callback : callbacks)
086            callback.endList();
087    }
088
089    /**
090     * Called when an atom is read.
091     *
092     * @param atom The atom bytes.
093     * @param displayHint The display hint bytes, if any were present.
094     * @throws ParseException If an atom is not appropriate given the current parser state.
095     */
096    protected final void onAtom(byte[] atom, Optional<byte[]> displayHint) throws ParseException
097    {
098        for (StreamingParserCallback callback : callbacks)
099            callback.onAtom(atom, displayHint);
100    }
101}
