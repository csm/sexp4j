001package org.metastatic.sexp4j;
002
003import java.util.*;
004
005/**
006 * A list of simpler expressions.
007 */
008public class ExpressionList extends AbstractList<Expression> implements Expression
009{
010    private final List<Expression> expressions;
011
012    /**
013     * Create a new, empty expression list.
014     */
015    public ExpressionList()
016    {
017        this.expressions = new ArrayList<>();
018    }
019
020    /**
021     * Create a new, empty expression list, with the given initial capacity.
022     *
023     * @param capacity The list capacity.
024     * @throws java.lang.IllegalArgumentException If the capacity is negative.
025     */
026    public ExpressionList(int capacity) {
027        this.expressions = new ArrayList<>(capacity);
028    }
029
030    /**
031     * Create a new, empty expression list, using the given list class as
032     * the underlying list representation.
033     *
034     * @param listClass The list implementation class; this class must provide
035     *                  a zero-argument constructor.
036     * @throws IllegalAccessException
037     * @throws InstantiationException
038     */
039    public ExpressionList(Class<? extends List<Expression>> listClass)
040            throws IllegalAccessException, InstantiationException
041    {
042        this.expressions = listClass.newInstance();
043    }
044
045    /**
046     * Create a new expression list containing the given expressions.
047     *
048     * @param expressions The expressions to add to the new list.
049     * @return The new list.
050     */
051    public static ExpressionList list(Expression... expressions) {
052        ExpressionList list = new ExpressionList();
053        Collections.addAll(list, expressions);
054        return list;
055    }
056
057    @Override
058    public Expression get(int index) {
059        return expressions.get(index);
060    }
061
062    @Override
063    public int size() {
064        return expressions.size();
065    }
066
067    @Override
068    public boolean add(Expression expression) {
069        return expressions.add(expression);
070    }
071
072    @Override
073    public void add(int index, Expression element) {
074        expressions.add(index, element);
075    }
076
077    @Override
078    public boolean addAll(int index, Collection<? extends Expression> c) {
079        return expressions.addAll(index, c);
080    }
081
082    @Override
083    public Iterator<Expression> iterator() {
084        return expressions.iterator();
085    }
086
087    @Override
088    public ListIterator<Expression> listIterator() {
089        return expressions.listIterator();
090    }
091
092    @Override
093    public ListIterator<Expression> listIterator(int index) {
094        return expressions.listIterator(index);
095    }
096
097    @Override
098    public List<Expression> subList(int fromIndex, int toIndex) {
099        return expressions.subList(fromIndex, toIndex);
100    }
101
102    @Override
103    public boolean equals(Object o) {
104        return (o instanceof ExpressionList) && expressions.equals(((ExpressionList) o).expressions);
105    }
106
107    @Override
108    public int hashCode() {
109        return expressions.hashCode();
110    }
111
112    @Override
113    public boolean contains(Object o) {
114        return expressions.contains(o);
115    }
116
117    @Override
118    public Object[] toArray() {
119        return expressions.toArray();
120    }
121
122    @Override
123    public <T> T[] toArray(T[] a) {
124        return expressions.toArray(a);
125    }
126
127    @Override
128    public boolean remove(Object o) {
129        return expressions.remove(o);
130    }
131
132    @Override
133    public boolean containsAll(Collection<?> c) {
134        return expressions.containsAll(c);
135    }
136
137    @Override
138    public boolean addAll(Collection<? extends Expression> c) {
139        return expressions.addAll(c);
140    }
141
142    @Override
143    public boolean removeAll(Collection<?> c) {
144        return expressions.removeAll(c);
145    }
146
147    @Override
148    public boolean retainAll(Collection<?> c) {
149        return expressions.retainAll(c);
150    }
151
152    @Override
153    public void clear() {
154        expressions.clear();
155    }
156
157    @Override
158    public int indexOf(Object o) {
159        return expressions.indexOf(o);
160    }
161
162    @Override
163    public int lastIndexOf(Object o) {
164        return expressions.lastIndexOf(o);
165    }
166
167    @Override
168    public Expression remove(int index) {
169        return expressions.remove(index);
170    }
171
172    @Override
173    public Expression set(int index, Expression element) {
174        return expressions.set(index, element);
175    }
176}
